async function seedDemo(){

async function fetchRows(){

  let rows=[];

  try{

    const res=await fetch(CSV_URL+'&t='+Date.now());

    const text=await res.text();

    text.trim().split('\n').slice(1).forEach(line=>{

      const cols=line.split(',');

      const name=(cols[0]||'').replace(/"/g,'').trim();

      if(!name)return;

      const ans=[];

      for(let i=0;i<TOTAL;i++) ans.push((cols[i+1]||'').replace(/"/g,'').trim().toUpperCase()||null);

      rows.push({name,ans});

    });

  }catch(e){}

  return rows;

}

  const tot=nA.length+nB.length||1;

  const ap=Math.round(nA.length/tot*100),bp=100-ap;

  const fa=isNina?'nina-a':'',fb=isNina?'nina-b':'b';

  const tag=(n)=>`<span class="name-tag${isNina?' nina':''}${n===playerName?' you':''}">${n}</span>`;

  return`<hr class="reveal-divider"/>

    <div class="reveal-label">How everyone voted so far</div>

    <div class="reveal-bar-row">

      <span class="reveal-bar-lbl">A</span>

      <div class="reveal-bar-track"><div class="reveal-bar-fill ${fa}" style="width:${ap}%"><span class="reveal-bar-text">${questions[qi].a}</span></div></div>

      <span class="reveal-pct">${ap}%</span>

    </div>

    ${nA.length?`<div class="reveal-names">${nA.map(tag).join('')}</div>`:''}

    <div class="reveal-bar-row" style="margin-top:4px;">

      <span class="reveal-bar-lbl">B</span>

      <div class="reveal-bar-track"><div class="reveal-bar-fill ${fb}" style="width:${bp}%"><span class="reveal-bar-text">${questions[qi].b}</span></div></div>

}

  else if(myAns&&!revealed) revealHTML=`<div class="loading-inline">Loading group results...</div>`;

  app.innerHTML=`<div class="card">

    ${q.nina?'<div class="nina-badge">Nina question</div>':''}

    <div class="q-label">Question ${current+1} of ${TOTAL}</div>

    <div class="q-text">${q.q}</div>

    ${q.sub?`<div class="q-sub">${q.sub}</div>`:''}

    <div class="options">

      <button class="opt-btn ${myAns==='A'?'sel':''}" onclick="sel('A')" ${myAns&&myAns!=='A'?'style="opacity:.6"':''}><span class="opt-label">A</span><br>${q.a}</button>

      <button class="opt-btn ${myAns==='B'?'sel':''}" onclick="sel('B')" ${myAns&&myAns!=='B'?'style="opacity:.6"':''}><span class="opt-label">B</span><br>${q.b}</button>

    </div>

    ${revealHTML}

  </div>

  <div class="nav">

    <button class="nav-btn" onclick="goBack()" ${current===0?'disabled':''}>Back</button>

    <span class="q-counter">${playerName} · ${current+1}/${TOTAL}</span>

    <button class="nav-btn primary" onclick="${isLast?'goToResults()':'nextQ()'}" ${!myAns||!revealed?'disabled':''}>${isLast?'See final results':'Next question'}</button>

  </div>`;

}


function nextQ(){revealed=false;cachedRows=null;current=Math.min(TOTAL-1,current+1);render();}

function goBack(){revealed=false;cachedRows=null;current=Math.max(0,current-1);render();}

function goToResults(){phase='results';render();}


function makeResultCard(rows,i,isNina){

  const q=questions[i];

  const nA=[],nB=[];

  rows.forEach(r=>{if(r.ans[i]==='A')nA.push(r.name);else if(r.ans[i]==='B')nB.push(r.name);});

  const tot=nA.length+nB.length||1;

  const ap=Math.round(nA.length/tot*100),bp=100-ap;

  const tag=n=>`<span class="name-tag${isNina?' nina':''}">${n}</span>`;

  return`<div class="result-card${isNina?' nina':''}">

    <div class="rc-question">${q.sub||q.q}</div>

    <div class="rc-bar-row"><span class="rc-bar-label">A</span>

      <div class="rc-bar-track"><div class="rc-bar-fill" style="width:${ap}%"><span class="rc-bar-text">${q.a}</span></div></div>

      <span class="rc-pct">${ap}%</span></div>

    ${nA.length?`<div class="rc-names">${nA.map(tag).join('')}</div>`:''}

    <div class="rc-bar-row" style="margin-top:4px;"><span class="rc-bar-label">B</span>

      <div class="rc-bar-track"><div class="rc-bar-fill b" style="width:${bp}%"><span class="rc-bar-text">${q.b}</span></div></div>

      <span class="rc-pct">${bp}%</span></div>

    ${nB.length?`<div class="rc-names">${nB.map(tag).join('')}</div>`:''}

    <div class="rc-voters">${nA.length+nB.length} vote${nA.length+nB.length!==1?'s':''}</div>

  </div>`;

}
